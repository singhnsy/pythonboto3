import sys

#Writing functions for add, sub & multiplication

def add(num1, num2):
    return num1 + num2

def sub(num1, num2):
    return num1 - num2

def mult(num1, num2):
    return num1 * num2

num1 = int(sys.argv[1])
operator = sys.argv[2]
num2 = int(sys.argv[3])

if operator == 'add':
    output = add(num1, num2)
    print (output)
elif operator == 'sub':
    output = sub(num1, num2)
    print (output)
elif operator == 'mult':
    output = mult(num1, num2)
    print(output)
else: 
    print("Please give the correct operator, it is add, sub, multi")

import sys

# Print all command-line arguments
print("Arguments passed:", sys.argv)

# First argument is always the script name
script_name = sys.argv[0]
print("Script name:", script_name)

# Other arguments are passed from index 1 onward
if len(sys.argv) > 1:
    first_argument = sys.argv[1]
    print("First argument:", first_argument)
else:
    print("No arguments passed.")
