# if-else-if, else
import sys
type = sys.argv[1]

if type == "t2.micro":
    print("It will charge you 2 dollar per day")

elif type == "t2.medium":
    print("It will charge you 4 dollar per day")

elif type == "t2.xlarge":
    print ("It will charge 8 dollars per day")

else:
    print("Please provide a valid instance type like t2.micro, t2.medium & t2.xlarge")
